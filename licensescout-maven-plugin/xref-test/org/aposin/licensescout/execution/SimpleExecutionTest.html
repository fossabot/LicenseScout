38      /**
39       * Test case for the method {@link Executor#execute()}.
40       * 
41       * @throws Exception
42       */
43      @Test
44      public void testExecutionJavaNoOutputs() throws Exception {
45          final File scanDirectory = new File("src/test/resources/scans/empty");
46          final ExecutionParameters executionParameters = createExecutionParametersNoOutputs(ArchiveType.JAVA,
47                  scanDirectory);
48          assertExecution(executionParameters);
49      }
50  
51      /**
52       * Test case for the method {@link Executor#execute()}.
53       * 
54       * @throws Exception
55       */
56      @Test
57      public void testExecutionJavaWithOutputs() throws Exception {
58          final File scanDirectory = new File("src/test/resources/scans/empty");
59          final ArrayList<Output> outputs = new ArrayList<>();
60          outputs.add(createOutput(OutputFileType.CSV, "licensereport.csv"));
61          outputs.add(createOutput(OutputFileType.HTML, "licensereport.html"));
62          outputs.add(createOutput(OutputFileType.TXT, "licensereport.txt"));
63          final ExecutionParameters executionParameters = createExecutionParameters(ArchiveType.JAVA, scanDirectory,
64                  outputs);
65          assertExecution(executionParameters);
66      }

69       * Test case for the method {@link Executor#execute()}.

71       * @throws Exception

73      @Test(expected = LicenseScoutExecutionException.class)
74      public void testExecutionJavaNoScanDirectory() throws Exception {
75          final File scanDirectory = null;
76          final ExecutionParameters executionParameters = createExecutionParametersNoOutputs(ArchiveType.JAVA,

88          final File scanDirectory = new File("not_existing");

94      /**
95       * Test case for the method {@link Executor#execute()}.

97       * @throws Exception
98       */
99      @Test
100     public void testExecutionNpmNoOutput() throws Exception {
101         final File scanDirectory = new File("src/test/resources/scans/empty");
102         final ExecutionParameters executionParameters = createExecutionParametersNoOutputs(ArchiveType.JAVASCRIPT,
103                 scanDirectory);
104         assertExecution(executionParameters);
105     }
106 
107     private void assertExecution(final ExecutionParameters executionParameters) throws LicenseScoutExecutionException {
108         final ConfigFileHandler configFileHandler = new FilesystemConfigFileHandler(new ConfigFileParameters(),
109                 TestUtil.createTestLog());
110         final Executor executor = new Executor(executionParameters, configFileHandler);
111         executor.execute();
112     }
113 
114     private ExecutionParameters createExecutionParametersNoOutputs(final ArchiveType archiveType,
115                                                                    final File scanDirectory) {
116         final ArrayList<Output> outputs = new ArrayList<>();
117         return createExecutionParameters(archiveType, scanDirectory, outputs);
118     }
119 
120     private ExecutionParameters createExecutionParameters(final ArchiveType archiveType, final File scanDirectory,
121                                                           final ArrayList<Output> outputs) {
122         final ExecutionParameters executionParameters = new ExecutionParameters();
123         executionParameters.setArchiveType(archiveType);
124         executionParameters.setFilteredVendorNames(new ArrayList<>());
125         executionParameters.setScanDirectory(scanDirectory);
126         executionParameters.setOutputDirectory(new File("target"));
127         executionParameters.setOutputs(outputs);
128         executionParameters.setCleanOutputActive(false);
129         // NOTE: exception occurs if the following is not set
130         executionParameters.setCleanOutputLegalStates(new LegalStatus[0]);
131         executionParameters.setLsLog(TestUtil.createTestLog());
132         executionParameters.setNpmExcludedDirectoryNames(new ArrayList<>());
133         executionParameters.setExporterFactories(Arrays.asList(new StandardReportExporterFactory()));
134         return executionParameters;
135     }
136 
137     private Output createOutput(final OutputFileType outputFileType, final String filename) {
138         final Output output = new Output();
139         output.setType(outputFileType);
140         output.setFilename(filename);
141         return output;
142     }
