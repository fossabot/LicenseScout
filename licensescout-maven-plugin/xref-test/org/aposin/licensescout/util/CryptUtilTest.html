16  package org.aposin.licensescout.util;

18  import java.io.File;
19  import java.io.FileInputStream;
20  import java.io.IOException;
21  import java.io.InputStream;
22  
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Tests for {@link CryptUtil}.
28   * 
29   */
30  public class CryptUtilTest {
31  
32      private static final String EXPECTED_MESSAGE_DIGEST_STRING = "83A2CA29DBAA80C0D6B6369448D0EDC0ACF30CA6E35C86F1E4C43A4915179F0D";
33  
34      /**
35       * Test for {@link CryptUtil#calculateMessageDigest(InputStream)}.
36       * 
37       * @throws IOException
38       */
39      @Test
40      public void testCalculateMessageDigest() throws IOException {
41          final File file = new File("src/test/resources/licensetexts/apache2/LICENSE-2.0-without-newlines.txt");
42          try (final FileInputStream fis = new FileInputStream(file)) {
43              final byte[] result = CryptUtil.calculateMessageDigest(fis).getBytes();
44              final String referenceString = EXPECTED_MESSAGE_DIGEST_STRING;
45              final byte[] reference = getByteArrayFromHexString(referenceString);
46              Assert.assertArrayEquals(reference, result);
47          }
48      }
49  
50      private byte[] getByteArrayFromHexString(final String hexString) {
51          final int numBytes = hexString.length() / 2;
52          final byte[] bytes = new byte[numBytes];
53          for (int i = 0; i < numBytes; i++) {
54              String s = hexString.substring(i * 2, i * 2 + 2);
55              byte b = (byte) Integer.parseUnsignedInt(s, 16);
56              bytes[i] = b;
57          }
58          return bytes;
59      }

61  }
