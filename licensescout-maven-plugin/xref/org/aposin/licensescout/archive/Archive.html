16  package org.aposin.licensescout.archive;

18  import java.util.ArrayList;
19  import java.util.Collection;
20  import java.util.List;
21  import java.util.Set;
22  
23  import org.apache.commons.collections4.MultiValuedMap;
24  import org.apache.commons.collections4.multimap.ArrayListValuedHashMap;
25  import org.aposin.licensescout.license.DetectionStatus;
26  import org.aposin.licensescout.license.LegalStatus;
27  import org.aposin.licensescout.license.License;
28  import org.aposin.licensescout.model.Author;
29  import org.aposin.licensescout.model.LSMessageDigest;
30  import org.aposin.licensescout.model.LicenseText;
31  import org.aposin.licensescout.model.Notice;
32  import org.aposin.licensescout.model.Provider;
33  import org.aposin.licensescout.util.MiscUtil;
34  
35  /**
36   * Representation of an archive or other artifact found during scanning.
37   * 
38   */
39  public class Archive implements Comparable<Archive> {
40  
41      private final ArchiveType archiveType;
42  
43      private final String fileName;
44  
45      private final String version;
46  
47      private final String path;
48  
49      private final MultiValuedMap<License, String> licenseList = new ArrayListValuedHashMap<>();
50      private List<License> detectedLicenses;
51  
52      private DetectionStatus detectionStatus;
53  
54      private LegalStatus legalStatus;
55  
56      private LSMessageDigest messageDigest;
57  
58      private final List<String> licenseCandidateFiles = new ArrayList<>();
59  
60      private String vendor;
61  
62      private String documentationUrl;
63  
64      private Provider provider;
65      private Notice notice;
66  
67      // NOTE: author is currently not used
68      private Author author;
69      private LicenseText licenseText;
70  
71      /**
72       * Instantiates a new archive.
73       * @param archiveType the type of the archive
74       * @param fileName the file name
75       * @param version the version of the archive
76       * @param path the path
77       */
78      public Archive(final ArchiveType archiveType, final String fileName, final String version, final String path) {
79          super();
80          this.archiveType = archiveType;
81          this.fileName = fileName;
82          this.version = version;
83          this.path = path;
84      }
85  
86      /**
87       * @return the archiveType
88       */
89      public final ArchiveType getArchiveType() {
90          return archiveType;
91      }
92  
93      /**
94       * Gets the file name.
95       * 
96       * @return the file name
97       */
98      public final String getFileName() {
99          return fileName;
100     }
101 
102     /**
103      * @return the version
104      */
105     public final String getVersion() {
106         return version;
107     }
108 
109     /**
110      * Gets the path.
111      * 
112      * @return the path
113      */
114     public final String getPath() {
115         return path;
116     }
117 
118     /**
119      * Adds a license
120      * @param license a license object
121      * @param path the path to the file containing the license
122      */
123     public void addLicense(final License license, final String path) {
124         licenseList.put(license, path);
125     }
126 
127     /**
128      * Clears the list of licenses.
129      */
130     public void clearLicenses() {
131         licenseList.clear();
132     }
133 
134     /**
135      * Gets the licenses.
136      * 
137      * @return the licenses
138      */
139     public Set<License> getLicenses() {
140         return licenseList.keySet();
141     }
142 
143     /**
144      * Gets the file paths.
145      * 
146      * @param license the license
147      * @return the file paths
148      */
149     public Collection<String> getFilePaths(final License license) {
150         return licenseList.get(license);
151     }
152 
153     /**
154      * @return the detectedLicenses
155      */
156     public final List<License> getDetectedLicenses() {
157         return detectedLicenses;
158     }
159 
160     /**
161      * @param detectedLicenses the detectedLicenses to set
162      */
163     public final void setDetectedLicenses(List<License> detectedLicenses) {
164         this.detectedLicenses = detectedLicenses;
165     }
166 
167     /**
168      * @return the detectionStatus
169      */
170     public final DetectionStatus getDetectionStatus() {
171         return detectionStatus;
172     }
173 
174     /**
175      * @param detectionStatus the detectionStatus to set
176      */
177     public final void setDetectionStatus(DetectionStatus detectionStatus) {
178         this.detectionStatus = detectionStatus;
179     }
180 
181     /**
182      * @return the legalStatus
183      */
184     public final LegalStatus getLegalStatus() {
185         return legalStatus;
186     }
187 
188     /**
189      * @param legalStatus the legalStatus to set
190      */
191     public final void setLegalStatus(LegalStatus legalStatus) {
192         this.legalStatus = legalStatus;
193     }
194 
195     /**
196      * @return the messageDigest
197      */
198     public final LSMessageDigest getMessageDigest() {
199         return messageDigest;
200     }
201 
202     /**
203      * @param messageDigest the messageDigest to set
204      */
205     public final void setMessageDigest(final LSMessageDigest messageDigest) {
206         this.messageDigest = messageDigest;
207     }
208 
209     /**
210      * @return the messageDigest
211      */
212     public final String getMessageDigestString() {
213         if (messageDigest != null) {
214             return MiscUtil.getHexString(messageDigest.getBytes());
215         } else {
216             return "";
217         }
218     }
219 
220     /**
221      * @return the licenseCandidateFiles
222      */
223     public final List<String> getLicenseCandidateFiles() {
224         return licenseCandidateFiles;
225     }
226 
227     /**
228      * @param filePath
229      */
230     public final void addLicenseCandidateFile(final String filePath) {
231         licenseCandidateFiles.add(filePath);
232     }
233 
234     /**
235      * @return the vendor
236      */
237     public final String getVendor() {
238         return vendor;
239     }
240 
241     /**
242      * @param vendor the vendor to set
243      */
244     public final void setVendor(String vendor) {
245         this.vendor = vendor;
246     }
247 
248     /**
249      * @return the documentationUrl
250      */
251     public final String getDocumentationUrl() {
252         return documentationUrl;
253     }
254 
255     /**
256      * @param documentationUrl the documentationUrl to set
257      */
258     public final void setDocumentationUrl(final String documentationUrl) {
259         this.documentationUrl = documentationUrl;
260     }
261 
262     /**
263      * @return the provider
264      */
265     public final Provider getProvider() {
266         return provider;
267     }
268 
269     /**
270      * @param provider the provider to set
271      */
272     public final void setProvider(Provider provider) {
273         this.provider = provider;
274     }
275 
276     /**
277      * @return the notice
278      */
279     public final Notice getNotice() {
280         return notice;
281     }
282 
283     /**
284      * @param notice the notice to set
285      */
286     public final void setNotice(Notice notice) {
287         this.notice = notice;
288     }
289 
290     /**
291      * @return the author
292      */
293     public final Author getAuthor() {
294         return author;
295     }
296 
297     /**
298      * @param author the author to set
299      */
300     public final void setAuthor(Author author) {
301         this.author = author;
302     }
303 
304     /**
305      * @return the licenseText
306      */
307     public final LicenseText getLicenseText() {
308         return licenseText;
309     }
310 
311     /**
312      * @param licenseText the licenseText to set
313      */
314     public final void setLicenseText(LicenseText licenseText) {
315         this.licenseText = licenseText;
316     }
317 
318     /**
319      * {@inheritDoc}
320      */
321     @Override
322     public int compareTo(final Archive other) {
323         if (this == other) {
324             return 0;
325         }
326         return this.getFileName().compareToIgnoreCase(other.getFileName());
327     }
328 
329     /**
330      * {@inheritDoc}
331      */
332     @Override
333     public boolean equals(final Object other) {
334         if (this == other) {
335             return true;
336         }
337         if (other instanceof Archive) {
338             return compareTo((Archive) other) == 0;
339         }
340         return false;
341     }
342 
343     /**
344      * {@inheritDoc}
345      */
346     @Override
347     public String toString() {
348         StringBuilder builder = new StringBuilder();
349         builder.append("Archive [archiveType=");
350         builder.append(archiveType);
351         builder.append(", fileName=");
352         builder.append(fileName);
353         builder.append(", version=");
354         builder.append(version);
355         builder.append(", path=");
356         builder.append(path);
357         builder.append(", licenseList=");
358         builder.append(licenseList);
359         builder.append(", detectionStatus=");
360         builder.append(detectionStatus);
361         builder.append(", legalStatus=");
362         builder.append(legalStatus);
363         builder.append(", messageDigest=");
364         builder.append(messageDigest);
365         builder.append("]");
366         return builder.toString();
367     }
