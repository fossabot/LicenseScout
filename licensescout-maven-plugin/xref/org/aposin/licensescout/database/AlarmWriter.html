16  package org.aposin.licensescout.database;
17  
18  import java.sql.Connection;
19  import java.sql.SQLException;
20  import java.sql.Statement;
21  import java.util.List;
22  
23  import org.aposin.licensescout.archive.Archive;
24  import org.aposin.licensescout.configuration.BuildInfo;
25  import org.aposin.licensescout.configuration.DatabaseConfiguration;
26  import org.aposin.licensescout.model.Alarm;
27  import org.aposin.licensescout.util.ILFLog;
28  
29  /**
30   * Writes alarms to a database.
31   * 
32   * @see Alarm
33   *
34   */
35  public class AlarmWriter {
36  
37      private AlarmWriter() {
38          // EMPTY
39      }
40  
41      /**
42       * Writes alarms to the database.
43       * 
44       * @param buildInfo
45       * @param alarms list of alarm objects
46       * @param databaseConfiguration
47       * @param log
48       */
49      public static void writeAlarms(final BuildInfo buildInfo, final List<Alarm> alarms,
50                                     final DatabaseConfiguration databaseConfiguration, final ILFLog log) {
51          try (final Connection connection = DatabaseUtil.getConnection(databaseConfiguration);
52                  final Statement statement = connection.createStatement();) {
53              for (final Alarm alarm : alarms) {
54                  final StringBuilder sb = new StringBuilder();
55                  sb.append("INSERT INTO Alarms ");
56                  sb.append("VALUES ('");
57                  sb.append(buildInfo.getName());
58                  sb.append("',')");
59                  sb.append(buildInfo.getVersion());
60                  sb.append("',')");
61                  sb.append(buildInfo.getDate());
62                  sb.append("',')");
63                  sb.append(alarm.getAlarmCause().name());
64                  sb.append("',')");
65                  final Archive archive = alarm.getArchive();
66                  sb.append(archive.getFileName());
67                  sb.append("')");
68                  final String sql = sb.toString();
69                  statement.executeUpdate(sql);
70              }
71          } catch (SQLException e) {
72              log.error(e);
73          }
74      }
75  }
