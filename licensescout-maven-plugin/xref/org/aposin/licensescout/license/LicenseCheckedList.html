60  /**
61   * Contains a map of archives with manually assigned licenses.
62   * 
63   * <p>This map is used if a license cannot be detected automatically. The archive is identified either by name and version or by name and message digest</p>
64   * 
65   * <p>Note that this class contains a hard-coded constant for the length of the message digest values. This should correspond to the algorithm returned by {@link CryptUtil#getMessageDigestAlgorithm()}.
66   * But while the algorithm can be changed, but this length of the message digests is currently fixed.</p>
67   * 
68   * @see LicenseUtil#evaluateLicenses(LicenseCheckedList, Archive, LicenseStoreData)
69   * @see CryptUtil#getMessageDigestAlgorithm()
70   */
71  public class LicenseCheckedList {

73      private static final int TYPE_INDEX = 0;
74      private static final int NAME_INDEX = 1;
75      private static final int VERSION_OR_MD_INDEX = 2;
76      private static final int DOC_URL_INDEX = 3;
77      private static final int PROVIDER_INDEX = 4;
78      private static final int NOTICE_INDEX = 5;
79      private static final int LICENSE_NAMES_OFFSET = 6;

82       * Note: this value is valid for SHA-256, should corresponds to the algorithm returned by {@link CryptUtil#getMessageDigestAlgorithm()}.
83       * (Note that the algorithm can be changed by the method {@link CryptUtil#setMessageDigestAlgorithm(String)}, but this length of the hash is always hard-coded.)
84       */
85      private static final int MESSAGE_DIGEST_NUM_BYTES = 32;
86      private static final int MESSAGE_DIGEST_NUM_CODED_CHARACTERS = MESSAGE_DIGEST_NUM_BYTES * 2;

98      /**
99       * Reads a CSV file containing checked archives.
100      * 
101      * @param inputStream a filename of the file to read
102      * @param licenseStoreData 
103      * @param providers 
104      * @param notices 
105      * @param log the logger
106      * @throws IOException if an error occurred while reading from the file
107      */
108     public void readCsv(final InputStream inputStream, final LicenseStoreData licenseStoreData,
109                         final Providers providers, final Notices notices, final ILFLog log)
110             throws IOException {
111         String line = "";
112         String cvsSplitBy = ",";
113         int lineNumber = 0;
114         try (final BufferedReader br = new BufferedReader(new InputStreamReader(inputStream))) {
115 
116             while ((line = br.readLine()) != null) {
117                 lineNumber++;
118                 // ignore lines commented out, empty lines and the header line
119                 if (line.startsWith("#") || StringUtils.isEmpty(line) || line.startsWith("Type")) {
120                     continue;
121                 }
122                 ArchiveIdentifier archiveIdentifier = null;
123                 // use comma as separator
124                 final String[] values = line.split(cvsSplitBy);
125                 final ArchiveType archiveType = getArchiveType(values[TYPE_INDEX].trim());
126                 final String name = values[NAME_INDEX].trim();
127                 // either version or hash
128                 final String id = values[VERSION_OR_MD_INDEX].trim();
129                 if (name.length() >= 0 && name.charAt(0) == '=') {
130                     PatternType patternType;
131                     int patternStartIndex;
132                     if (name.length() >= 1 && name.charAt(1) == '=') {
133                         patternType = PatternType.PATTERN_ON_PATH;
134                         patternStartIndex = 2;
135                     } else {
136                         patternType = PatternType.PATTERN_ON_FILENAME;
137                         patternStartIndex = 1;
138                     }
139                     final String regex = name.substring(patternStartIndex);
140                     archiveIdentifier = new ArchiveIdentifierPattern(archiveType, patternType, regex);
141                 } else if (id.length() >= 0) {
142                     if (id.length() == MESSAGE_DIGEST_NUM_CODED_CHARACTERS) {
143                         archiveIdentifier = new ArchiveIdentifierMessageDigest(archiveType, name, id);
144                     } else {
145                         archiveIdentifier = new ArchiveIdentifierVersion(archiveType, name, id);
146                     }
147                 } else {
148                     log.error("read line with error, no version or message digest");
149                     archiveIdentifier = null;
150                 }
151                 final String documentationUrl = values[DOC_URL_INDEX].trim();
152                 final String providerId = values[PROVIDER_INDEX].trim();
153                 final int lineNumberFinal = lineNumber;
154                 final Provider provider = getProviderFromId(providers, providerId, log,
155                         () -> "line " + lineNumberFinal);
156                 final String noticeId = values[NOTICE_INDEX].trim();
157                 final Notice notice = getNoticeFromId(notices, noticeId);
158 
159                 if (archiveIdentifier != null) {
160                     final LicenseResult licenseResult = fetchLicenseResult(licenseStoreData, log, values,
161                             documentationUrl, notice, provider);
162                     if (archiveIdentifier.getNameMatchingType() == NameMatchingType.EXACT) {
163                         manualArchives.put(archiveIdentifier, licenseResult);
164                     } else {
165                         manualPatternArchives.put((ArchiveIdentifierPattern) archiveIdentifier, licenseResult);
166                     }
167                 }
168             }
169         }
170     }

172     /**
173      * Reads an XML file containing checked archives.
174      * 
175      * Note: currently not used.
176      * 
177      * @param file a filename of the file to read
178      * @param licenseStoreData 
179      * @param notices 
180      * @param providers 
181      * @param validateXml 
182      * @param log the logger
183      * @throws IOException if an error occurred while reading from the file
184      * @throws ParserConfigurationException 
185      * @throws SAXException 
186      */
187     public void readXml(final File file, final LicenseStoreData licenseStoreData, final Notices notices,
188                         final Providers providers, boolean validateXml, final ILFLog log)
189             throws IOException, ParserConfigurationException, SAXException {
190 
191         final SAXParserFactory spf = SAXParserFactory.newInstance();
192         spf.setNamespaceAware(true);
193         spf.setValidating(validateXml);
194         final SAXParser saxParser = spf.newSAXParser();
195         final XMLReader xmlReader = saxParser.getXMLReader();
196         xmlReader.setContentHandler(new ArchiveSaxHandler(licenseStoreData, notices, providers, log));
197         xmlReader.parse(file.toURI().toString());
198     }

200     /**
201      * @param licenseStoreData
202      * @param log
203      * @param values
204      * @param documentationUrl 
205      * @return a result object
206      */
207     private LicenseResult fetchLicenseResult(final LicenseStoreData licenseStoreData, final ILFLog log,
208                                              final String[] values, final String documentationUrl, final Notice notice,
209                                              final Provider provider) {
210         final int numLicenseNames = values.length - LICENSE_NAMES_OFFSET;
211         final List<License> licenses = new ArrayList<>();
212         if (numLicenseNames == 0) {
213             licenses.add(licenseStoreData.getLicenseBySpdxIdentifier(LicenseSpdxIdentifier.NO_MANUAL_INFORMATION));
214         } else {
215             for (int i = 0; i < numLicenseNames; i++) {
216                 final String licenseName = values[i + LICENSE_NAMES_OFFSET].trim();
217                 final License license = licenseStoreData.getLicenseBySpdxIdentifier(licenseName);
218                 if (license == null) {
219                     log.error("License not found in store: '" + licenseName + "'");
220                 } else {
221                     licenses.add(license);
222                 }
223             }
224         }
225         return new LicenseResult(licenses, documentationUrl, notice, provider);
226     }

299     /**
300      * @return the collected CSV contents
301      */
302     private static String writeCsvSkeletonFile(final List<Archive> archives) {
303         final StringBuilder sb = new StringBuilder();
304         final String nl = "\n";
305         // header line
306         sb.append(
307                 "Type,Filename or Regex,Version or Message digest,Documentation URL, Provider,Notice,License 1,License 2,License 3")
308                 .append(nl);
309         // body
310         for (final Archive archive : archives) {
311             sb.append(archive.getArchiveType().name()).append(",");
312             sb.append(archive.getFileName()).append(",");
313             if (!StringUtils.isEmpty(archive.getVersion())) {
314                 sb.append(archive.getVersion());
315             } else if (archive.getMessageDigest() != null) {
316                 sb.append(archive.getMessageDigestString());
317             } else {
318                 sb.append("TODO");
319             }
320             sb.append(",");
321             sb.append("TODO_docURL,"); // documentation URL
322             sb.append("TODO_provider,"); // provider
323             sb.append("TODO_notice,"); // notice
324             sb.append("TODO_license1");
325             sb.append(nl);
326         }
327         return sb.toString();
328     }

344     /**
345      * @return the collected XML string
346      */
347     private static String writeXmlSkeletonFile(final List<Archive> archives) {
348         final StringBuilder sb = new StringBuilder();
349         final String nl = "\n";
350         beginLine(sb, 0).append("<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\" ?>").append(nl);
351         beginLine(sb, 0).append("<archives>").append(nl);
352         for (final Archive archive : archives) {
353             beginLine(sb, 1).append("<archive>").append(nl);
354             beginLine(sb, 2).append("<archiveType>").append(archive.getArchiveType().name()).append("</archiveType>")
355                     .append(nl);
356             beginLine(sb, 2).append("<name>").append(archive.getFileName()).append("</name>").append(nl);
357             if (!StringUtils.isEmpty(archive.getVersion())) {
358                 beginLine(sb, 2).append("<version>").append(archive.getVersion()).append("</version>").append(nl);
359             } else if (archive.getMessageDigest() != null) {
360                 beginLine(sb, 2).append("<messageDigest algorithm='").append(CryptUtil.getMessageDigestAlgorithm())
361                         .append("'>").append(archive.getMessageDigestString()).append("</messageDigest>").append(nl);
362             }
363             beginLine(sb, 2).append("<documentationUrl>").append("TODO").append("</documentationUrl>").append(nl);
364             beginLine(sb, 2).append("<provider>").append("TODO").append("</provider>").append(nl);
365             beginLine(sb, 2).append("<notice>").append("TODO").append("</notice>").append(nl);
366             beginLine(sb, 2).append("<licenses>").append(nl);
367             beginLine(sb, 3).append("<license>").append("TODO").append("</license>").append(nl);
368             beginLine(sb, 2).append("</licenses>").append(nl);
369             beginLine(sb, 1).append("</archive>").append(nl);
370         }
371         beginLine(sb, 0).append("</archives>").append(nl);
372         return sb.toString();
373     }

375     private static StringBuilder beginLine(final StringBuilder sb, final int indentationLeve) {
376         final String indentationString = StringUtils.repeat("\t", indentationLeve);
377         sb.append(indentationString);
378         return sb;
379     }

386     private static Provider getProviderFromId(final Providers providers, final String providerId, final ILFLog log,
387                                               final Supplier<String> context) {
388         final Provider provider = providers.getProviderByIdentifier(providerId);
389         if (provider == null) {
390             String message = "Cannot find provider with ID: '" + providerId + "'";
391             if (context != null && !StringUtils.isEmpty(context.get())) {
392                 message += " (" + context.get() + ")";
393             }
394             log.warn(message);
395         }
396         return provider;
397     }

467         private class StandardArchiveHandler implements IElementHandler {
468 
469             /**
470              * {@inheritDoc}
471              */
472             @Override
473             public String getElementName() {
474                 return ELEMENT_ARCHIVE;
475             }
476 
477             /** {@inheritDoc} */
478             @Override
479             public void startElement(String uri, String localName, String qName, final Attributes attributes) {
480                 archiveType = null;
481                 name = null;
482                 patternType = null;
483                 version = null;
484                 messageDigest = null;
485                 licenses = new ArrayList<>();
486                 documentationUrl = null;
487                 notice = null;
488                 provider = null;
489             }
490 
491             /** {@inheritDoc} */
492             @Override
493             public void endElement(String uri, String localName, String qName) {
494                 ArchiveIdentifier archiveIdentifier = null;
495                 if (archiveType == null) {
496                     getLog().warn("No archive type for archive! Line: " + getLineNumberInformation());
497                 }
498                 if (name == null) {
499                     getLog().warn("No name for archive! Line: " + getLineNumberInformation());
500                 }
501                 if (patternType != null) {
502                     archiveIdentifier = new ArchiveIdentifierPattern(archiveType, patternType, name);
503                 } else {
504                     if (version != null) {
505                         archiveIdentifier = new ArchiveIdentifierVersion(archiveType, name, version);
506                     }
507                     if (messageDigest != null) {
508                         archiveIdentifier = new ArchiveIdentifierMessageDigest(archiveType, name, messageDigest);
509                     }
510                 }
511 
512                 if (archiveIdentifier != null) {
513                     final LicenseResult licenseResult = fetchLicenseResult(licenses, documentationUrl, notice,
514                             provider);
515                     if (archiveIdentifier.getNameMatchingType() == NameMatchingType.EXACT) {
516                         manualArchives.put(archiveIdentifier, licenseResult);
517                     } else {
518                         manualPatternArchives.put((ArchiveIdentifierPattern) archiveIdentifier, licenseResult);
519                     }
520                 }
521             }
522         }

552              * {@inheritDoc}

554             @Override
555             protected void startElementHook(final String uri, final String localName, final String qName,
556                                             final Attributes attributes) {
557                 final String typeText = attributes.getValue(ATTRIBUTE_TYPE);
558                 if (typeText != null) {
559                     switch (typeText) {
560                         case "exact":
561                             patternType = null;
562                             break;

566                         case "regexpath":
567                             patternType = PatternType.PATTERN_ON_FILENAME;
568                             break;
569                         default:
570                             getLog().warn("Unknown matching type: '" + typeText + "'");
571                             break;
572                     }
573                 }
574             }
575 
576             /** {@inheritDoc} */
577             @Override
578             protected void processText(final String text) {
579                 name = text;
580             }
