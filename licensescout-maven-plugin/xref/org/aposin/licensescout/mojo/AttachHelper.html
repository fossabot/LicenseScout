16  package org.aposin.licensescout.mojo;
17  
18  import java.io.File;
19  
20  import org.apache.maven.project.MavenProject;
21  import org.apache.maven.project.MavenProjectHelper;
22  import org.aposin.licensescout.configuration.Output;
23  import org.aposin.licensescout.execution.ExecutionParameters;
24  import org.aposin.licensescout.util.OutputFileHelper;
25  
26  /**
27   * Helper class for {@link AbstractScanMojo}.
28   *
29   */
30  public class AttachHelper {
31  
32      /**
33       * Attaches generated report files as secondary artifacts.
34       * 
35       * @param mavenProject 
36       * @param mavenProjectHelper 
37       * @param executionParameters 
38       * 
39       */
40      public static void attachReports(final MavenProject mavenProject, final MavenProjectHelper mavenProjectHelper,
41                                       final ExecutionParameters executionParameters) {
42          final String artifactClassifier = "licensereport";
43          for (final Output output : executionParameters.getOutputs()) {
44              final String artifactType = output.getType().getArtifactType();
45              final File artifactFile = new File(executionParameters.getOutputDirectory(),
46                      OutputFileHelper.getOutputFilename(output));
47              executionParameters.getLsLog().info("attaching artifact: " + artifactFile.getAbsolutePath());
48              mavenProjectHelper.attachArtifact(mavenProject, artifactType, artifactClassifier, artifactFile);
49          }
50      }
51  }
