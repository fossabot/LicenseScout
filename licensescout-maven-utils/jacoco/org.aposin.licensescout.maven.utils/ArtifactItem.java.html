package org.aposin.licensescout.maven.utils;

import org.apache.maven.plugins.annotations.Parameter;

/**
 * Description of an artifact by Maven coordinates for use in Maven configuration.
 */
public class ArtifactItem {

    /**
     * Group Id of artifact.
     */
    @Parameter(property = "groupId", required = true)
    private String groupId;

    /**
     * Artifact ID of artifact.
     */
    @Parameter(property = "artifactId", required = true)
    private String artifactId;

    /**
     * Version of artifact.
     */
    @Parameter(property = "version", required = true)
    private String version;

    /**
     * Type of Artifact (War,Jar,etc)
     *
     * @parameter
     * @required
     */
    @Parameter(property = "type", required = false)
    private String type;

    /**
     * Classifier for Artifact (tests,sources,etc)
     *
     * @parameter
     */
    @Parameter(property = "classifier", required = false)
    private String classifier;

    /**
     * Constructor used by the Maven runtime system.
     */
    public ArtifactItem() {
        // EMPTY
    }

    /**
     * Constructor for testing.
     * @param groupId
     * @param artifactId
     * @param version
     * @param type
     * @param classifier
     */
    public ArtifactItem(String groupId, String artifactId, String version, String type, String classifier) {
        this.groupId = groupId;
        this.artifactId = artifactId;
        this.version = version;
        this.type = type;
        this.classifier = classifier;
    }

    /**
     * @return the groupId
     */
    public final String getGroupId() {
        return groupId;
    }

    /**
     * @return the artifactId
     */
    public final String getArtifactId() {
        return artifactId;
    }

    /**
     * @return the version
     */
    public final String getVersion() {
        return version;
    }

    /**
     * @return the type
     */
    public final String getType() {
        return type;
    }

    /**
     * @return the classifier
     */
    public final String getClassifier() {
        return classifier;
    }

}
